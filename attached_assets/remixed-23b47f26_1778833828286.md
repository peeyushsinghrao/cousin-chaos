  return PageRouteBuilder<T>(
    transitionDuration: const Duration(milliseconds: 400),
    reverseTransitionDuration: const Duration(milliseconds: 300),
    pageBuilder: (_, __, ___) => page,
    transitionsBuilder: (_, animation, __, child) {
      return FadeTransition(
        opacity: CurvedAnimation(parent: animation, curve: Curves.easeOut),
        child: ScaleTransition(
          scale: Tween<double>(begin: 0.92, end: 1.0).animate(

          ),
          child: child,
        ),
      );

}

#### Phase 2 — Card Reveal (CRITICAL — HOLD MECHANIC)
The card must be HELD for 1 full second to flip. A quick tap does nothing.

```dart
// Hold mechanic implementation:
Timer? _holdTimer;
bool _isHolding = false;
bool _hasFlipped = false;

// In GestureDetector:
onLongPressStart: (_) {
  _isHolding = true;
  HapticFeedback.mediumImpact(); // cardReveal haptic
  _shakeController.repeat();    // card shakes while holding
  _holdTimer = Timer(const Duration(seconds: 1), () {
    if (_isHolding && !_hasFlipped && mounted) {
      _hasFlipped = true;
      _shakeController.stop();
      _shakeController.reset();
      _flipController.forward(); // 3D flip
    }
  });
},
onLongPressEnd: (_) {
  if (_hasFlipped) return;
  _isHolding = false;
  _holdTimer?.cancel();
  _shakeController.stop();
  _shakeController.reset();
},
onLongPressCancel: () {
  // same as end
},
```
