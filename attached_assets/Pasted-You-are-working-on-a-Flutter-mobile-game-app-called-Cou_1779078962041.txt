Create lib/services/sound_service.dart as a singleton:

  class SoundService {
    static final SoundService _instance = SoundService._internal();
    factory SoundService() => _instance;
    SoundService._internal();

    bool isMuted = false;
    final Map<String, AudioPlayer> _players = {};

    Future<void> init() async {
      // Preload all sounds at app start
      final sounds = [
        'card-flip', 'tap', 'countdown', 'timer_tick',
        'wheel_spin', 'card_reveal', 'freeze', 'win',
        'game_start', 'explosion'
      ];
      for (final s in sounds) {
        final player = AudioPlayer();
        await player.setSource(AssetSource('sounds/$s.mp3'));
        _players[s] = player;
      }
    }

    Future<void> play(String name, {double volume = 1.0}) async {
      if (isMuted) return;
      final player = _players[name];
      if (player == null) return;
      await player.stop();
      await player.setVolume(volume);
      await player.resume();
    }

    void toggleMute() => isMuted = !isMuted;
    void dispose() {
      for (final p in _players.values) p.dispose();
    }
  }
